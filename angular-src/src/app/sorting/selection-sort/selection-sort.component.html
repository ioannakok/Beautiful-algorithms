Java

void selectionSort(int arr[]) {

int n = arr.length;
for (int i = 0; i < n-1; i++) {
int min_idx = i;

for (int j = i+1; j < n; j++) {
if (arr[j] < arr[min_idx]){
min_idx = j;
int temp = arr[min_idx];
arr[min_idx] = arr[i];
arr[i] = temp;
}
}
}
}

Python

def selectionSort(arr):

for i in range(len(arr)):
min_idx = i

for j in range(i+1, len(arr)):
if arr[min_idx] > arr[j]:
min_idx = j
arr[i], arr[min_idx] = arr[min_idx], arr[i]
