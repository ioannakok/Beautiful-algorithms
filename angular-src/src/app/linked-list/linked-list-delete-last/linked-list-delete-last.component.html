Java

void deleteLast() {

if(listHead == null)
throw Exception("Linked list is empty");

if(listHead.next == null){
listHead = null;
listTail = null;
} else {
Node nodePointer = listHead;

while(nodePointer.next != listTail) {
nodePointer = nodePointer.next;
}
listTail = nodePointer;
}
}
