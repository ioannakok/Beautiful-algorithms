void bubbleSort(int arr[]) {

int n = arr.length;
for(int i=0; i < n; i++) {
for(int j=0; j < n-i-1; j++) {
if(arr[j] > arr[j+1]) {

arr[j] = arr[j+1];
arr[j+1] = temp;
}
}
}
}

Python

def bubbleSort(arr):
n = len(arr)

for i in range(n):
for j in range(0, n-i-1):
if arr[j] > arr[j+1]:
arr[j], arr[j+1] = arr[j+1], arr[j]
